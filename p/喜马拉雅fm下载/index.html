$(".sound-list ul li a").each(function(i){
    var sound_id = $(this).attr("sound_id");
    var json_url = "http://www.ximalaya.com/tracks/" + sound_id + ".json";
    $.get(json_url,function(json){
        var save_name = $(this).html;
        var down_cmd = "wget -c -T 3 -t 100 '" + json.play_path_64 + "' -O '" + save_name + "'";
        console.log(down_cmd);
    })
})

function getDownloadCommand(){
    let player = JSON.parse(localStorage.player_track);
    if(localStorage.download_playing_name == player.trackName){
        console.log('重复:wget '+player.trackName);
    } else {
        let down_cmd = 'wget -c -T 3 -t 100 "' + player.src + '" -O "' + player.trackName + '.m4a";'+"\r\n";
        localStorage.download_cmd+=down_cmd;
        localStorage.download_playing_name=player.trackName;
        console.log('wget '+player.trackName);
        index ++;
    }
}
playAudio();
